  <h1>Jogadores</h1>
  <div id="lista-jogadores"></div>

  <script>
    const sheetURL = 'https://docs.google.com/spreadsheets/d/1DSpzimTw_mpUiQKNYjRv-xsTjUkqrQJtj26trdZUfL8/edit?usp=sharing';

    Tabletop.init({
      key: sheetURL,
      simpleSheet: false,
      wanted: ["Jogadores"],
      callback: function(data, tabletop) {
        const jogadores = tabletop.sheets("Jogadores").elements;
        const container = document.getElementById("lista-jogadores");

        jogadores.forEach(j => {
          const div = document.createElement("div");
          div.className = "jogador";
          div.innerHTML = `
            <div class="jogador-line">
              <img src="${j.Foto}" alt="${j.Nome}">
              <strong>${j.Nome}</strong>
            </div>
          `;
          container.appendChild(div);
        });
      }
    });
  </script>
